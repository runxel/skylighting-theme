def power(x,n)
    result = 1
    while n.nonzero?
        if n[0].nonzero?
            result *= x
            n -= 1
        end
        x *= x
        n /= 2
    end
    return result
end

def f(x)
    Math.sqrt(x.abs) + 5*x ** 3
end
